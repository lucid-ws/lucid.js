    'use strict';

/**
* Object containing the options for the Server
* @typedef {(object)} ServerOptions
* @property {Number} [port=25544] The port that the WebSocket and HTTP server should run on.
* @property {ExpressServerOptions} [express] Options for the Express (REST API) Server.
* @example
* {
*   wss: {},
*   express: {},
*   port: 25544
* }
*/

import EventEmitter from 'events';
import ExpressServer from './ExpressServer';
import * as Util from '../Util';
import * as Constants from '../Constants';

/**
* Manages the Express (HTTP), WebSocket Server and connections
* @example
* // ES6
* import Server from 'lucid';
* let server = new Server({
*   port: 25544
* });
*
* // CommonJS
* var Lucid = require("lucid");
* var server = new Lucid.Server({
*   port: 25544
* });
*/
class LucidServer extends EventEmitter{

  /**
  * Creates a new LucidServer class.
  * @param  {ServerOptions} [options] Options to pass to the object.
  * @example
  * import Server from 'lucid';
  * var server = new Server({
  *   port: 25544
  * });
  *
  * // CommonJS
  * var Lucid = require("lucid");
  * var server = new Lucid.Server({
  *   port: 25544
  * });
  */
  constructor(options) {
    super();
    /**
    * The options passed to the Server upon instantiation.
    * @type {ServerOptions}
    */
    this.options = Util.Options(ServerConstants.ServerDefaultOptions, options);

    /**
     * The status of the server
     * @type {ServerStatus}
     * @example
     * import ServerConstants from 'lucid'
     * var available = (server.status === ServerConstants.Status.AVAILABLE);
     */
    this.status = ServerConstants.ServerStatus.UNAVAILABLE;

    if (this.options.autoStart) {
      this._setup();
    }
  }

  /**
   * Tries to start the server
   * @throws {SERVER_ACTIVE_ERROR} error when
   */
  start() {

  }

  /**
   * Sets up the Client's Express and WebSocket Servers.
   * @private
   */
  _setup() {
    this.status = ServerConstants.ServerStatus.STARTING;

    /**
     * The ExpressServer of the server - handles HTTP requests
     * @type {ExpressServer}
     * @private
     */
    this._expressServer = new ExpressServer(this);
  }

};

export default LucidServer;
